## redis 相关操作

### 连接

```go
package main

import (
    "fmt"
    "github.com/go-redis/redis"
)

func main() {
    client := redis.NewClient(&redis.Options{
        Addr:     "localhost:6379",
        Password: "", // no password set
        DB:       0,  // use default DB
    })

    pong, err := client.Ping().Result()
    fmt.Println(pong, err
    // Output: PONG <nil>
}

```

### 设置

```go
package main

import (
    "fmt"
    "github.com/go-redis/redis"
)

func main() {
    client := redis.NewClient(&redis.Options{
        Addr:     "localhost:6379",
        Password: "", // no password set
        DB:       0,  // use default DB
    })

    err := client.Set("key", "value", 0).Err()
    if err != nil {
        panic(err)
    }

    val, err := client.Get("key").Result
    if err != nil {
        panic(err)
    }
    fmt.Println("key", val)
    // Output: key value
}

```

### 过期

```go
package main

import (
    "fmt"
    "github.com/go-redis/redis"
)

func main() {
    client := redis.NewClient(&redis.Options{
        Addr:     "localhost:6379",
        Password: "", // no password set
        DB:       0,  // use default DB
    })

    err := client.Expire("key", 10*time.Second).Err()
    if err != nil {
        panic(err)
    }

    ttl, err := client.TTL("key").Result
    if err != nil {
        panic(err)
    }

}

```

### 列表

```go
package main

import (
    "fmt"
    "github.com/go-redis/redis"
)

func main() {
    client := redis.NewClient(&redis.Options{
        Addr:     "localhost:6379",
        Password: "", // no password set
        DB:       0,  // use default DB
    })

    err := client.LPush("languages", "go").Err()
    if err != nil {
        panic(err)
    }

    err = client.LPush("languages", "java").Err()
    if err != nil {
        panic(err)
    }

    val, err := client.LRange("languages", 0, -1).Result
    if err != nil {
        panic(err)
    }
    fmt.Println("languages", val)
    // Output: languages [java go]
}

```

### 集合

```go
package main

import (
    "fmt"
    "github.com/go-redis/redis"
)

func main() {
    client := redis.NewClient(&redis.Options{
        Addr:     "localhost:6379",
        Password: "", // no password set
        DB:       0,  // use default DB
    })

    err := client.SAdd("languages", "go").Err()
    if err != nil {
        panic(err)
    }

    err = client.SAdd("languages", "java").Err()
    if err != nil {
        panic(err)
    }

    val, err := client.SMembers("languages").Result
    if err != nil {
        panic(err)
    }
    fmt.Println("languages", val)
    // Output: languages [java go]
}

```
