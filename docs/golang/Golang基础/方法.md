# 方法

## 语法

```go
// 普通写法
func (r ReceiverType) funcName(parameters) (results) {
    <函数体>
}

func (r *ReceiverType) funcNamePtr(parameters) (results) {
    <函数体>
}

// 范型写法
func (r ReceiverType[T]) funcName(parameters) (results) {
    <函数体>
}

func (r *ReceiverType[T]) funcNamePtr(parameters) (results) {
    <函数体>
}
```

## 示例

::: tabs#fruit

@tab 普通方法#普通方法

- 普通方法

```go
package main

import "fmt"

type Person struct {
    name string
    age  int
}

func main() {
    p := Person{
        name: "张三",
        age:  18,
    }
    p.sayHello()
}

func (p Person) sayHello() {
    fmt.Printf("Hello, my name is %s, I'm %d years old.\n", p.name, p.age)
}
```

@tab 范型方法#范型方法

- 范型方法

```go
package main

import "fmt"

type Person[T any] struct {
	name  string
	age   int
	value T
}

func main() {
	p := Person[float64]{
		name:  "张三",
		age:   18,
		value: 1.2,
	}
	p.sayHello()
}

func (p Person[T]) sayHello() {
	fmt.Printf("Hello, my name is %s, I'm %d years old. value %v\n", p.name, p.age, p.value)
}

```

## 接口实现

```go
package main

import "fmt"

type Person struct {
    name string
    age  int
}

type Human interface {
    sayHello()
}

func main() {
    p := Person{
        name: "张三",
        age:  18,

    p.sayHello()

```

## 方法重写

```go
package main

import "fmt"

type Person struct {
    name string
    age  int
}

type Student struct {
    Person
    school string
}

func main() {
    s := Student{
        Person: Person{
            name: "张三",
            age:  18,
        },
        school: "清华大学",
    }
    s.sayHello()
}

func (p Person) sayHello() {
    fmt.Printf("Hello, my name is %s, I'm %d years old.\n", p.name, p.age)
}

func (s Student) sayHello() {
    fmt.Printf("Hello, my name is %s, I'm %d years old. I'm a student of %s.\n", s.name, s.age, s.school)
}

```
